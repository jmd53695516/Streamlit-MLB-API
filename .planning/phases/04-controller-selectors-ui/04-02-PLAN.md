    def build_view(
        team_id: int,
        player_id: int,
        venue_id: int,
        *,
        season: int | None = None,
        api: Any = None,
    ) -> ViewModel:
        """Compose services + pipeline + verdict matrix into a ViewModel.

        Pure function — no Streamlit, no session_state (D-23).
        Not cached (D-22) — caching lives at services boundary.
        """
        if api is None:
            from mlb_park.services import mlb_api as api  # local import, NOT streamlit
        season = season if season is not None else CURRENT_SEASON

        # 1. Lookups for selection-context fields
        teams = api.get_teams()
        team = next(t for t in teams if t["id"] == team_id)
        team_abbr = team["abbreviation"]
        player_home_venue_id = team["venue"]["id"]

        roster = api.get_team_hitting_stats(team_id, season)
        player_entry = next(
            e for e in roster if e["person"]["id"] == player_id
        )
        player_name = player_entry["person"]["fullName"]

        parks_dict = api.load_all_parks()             # dict[int, dict]
        venue_name = parks_dict[venue_id].get("name", "")

        # 2. Run Phase 3 pipeline
        result = extract_hrs(player_id, season=season, api=api)
        events = result.events  # tuple[HREvent, ...]
        errors = result.errors  # tuple[PipelineError, ...]

        # 3. Filter plottable (D-07)
        plottable = tuple(
            ev for ev in events if ev.has_distance and ev.has_coords
        )

        # 4. Verdict matrix (D-10: None if plottable empty)
        if plottable:
            hit_data_list = [hr_event_to_hit_data(ev) for ev in plottable]
            # hr_event_to_hit_data returns None when an event isn't plottable;
            # because we already filtered by has_distance & has_coords, none
            # should be None here — but keep the defensive filter.
            hit_data_list = [h for h in hit_data_list if h is not None]
            park_objs = load_parks(parks_dict)        # dict[int, Park], D-02 via pipeline re-export
            matrix = compute_verdict_matrix(hit_data_list, park_objs)
            # NB: compute_verdict_matrix(hrs, parks) — hrs FIRST, parks SECOND
            # (verified in src/mlb_park/geometry/verdict.py:96).
        else:
            matrix = None

        # 5. clears_selected_park (D-08) — positional alignment via dense cleared matrix
        clears = _clears_for_venue(matrix, venue_id) if matrix is not None else ()

        # 6. Totals (D-09)
        totals = _compute_totals(events, plottable, matrix)

        return ViewModel(
            season=season,
            team_id=team_id,
            team_abbr=team_abbr,
            player_id=player_id,
            player_name=player_name,
            venue_id=venue_id,
            venue_name=venue_name,
            player_home_venue_id=player_home_venue_id,
            events=events,

            totals=totals,
            errors=errors,

    **`_clears_for_venue` — exact implementation (D-08):**

    VerdictMatrix alignment is POSITIONAL. Row i of `matrix.cleared` corresponds
    to `matrix.hrs[i]`, which was the i-th HitData passed into
    `compute_verdict_matrix`. Since we built `hit_data_list` from `plottable`
    in the same order, `matrix.cleared[i, j]` tells us whether
    `plottable[i]` would clear the fence at `matrix.venue_ids[j]`.

    To get the boolean tuple for the SELECTED `venue_id`, find the column
    index for that venue in `venue_ids` and slice:

    ```python
    def _clears_for_venue(matrix: VerdictMatrix, venue_id: int) -> tuple[bool, ...]:
        """Return per-plottable-HR bool: does this HR clear the fence at `venue_id`?

        Alignment: result[i] corresponds to plottable_events[i] (positional,
        same order we fed into compute_verdict_matrix).

        Raises KeyError if venue_id not found in the matrix's venue_ids array
        (shouldn't happen — load_all_parks returns all 30 MLB venues).
        """
        venue_ids_list = matrix.venue_ids.tolist()
        try:
            j = venue_ids_list.index(int(venue_id))
        except ValueError as e:
            raise KeyError(f"venue_id {venue_id} not in verdict matrix") from e
        # matrix.cleared is (n_hrs, n_parks) bool; column j is this venue's verdict per HR
        return tuple(bool(x) for x in matrix.cleared[:, j])
    ```

    **`_compute_totals` (D-09):**

    ```python
    def _compute_totals(
        events: tuple,
        plottable: tuple,
        matrix: VerdictMatrix | None,
    ) -> dict[str, int | float]:
        n_plottable = len(plottable)
        if matrix is None or n_plottable == 0:
            return {
                "total_hrs": len(events),
                "plottable_hrs": n_plottable,
                "avg_parks_cleared": 0.0,
                "no_doubters": 0,
                "cheap_hrs": 0,
            }
        # Per-HR count of parks cleared (sum across cols).
        per_hr_cleared = matrix.cleared.sum(axis=1)   # shape (n_hrs,)
        return {
            "total_hrs": len(events),
            "plottable_hrs": n_plottable,
            "avg_parks_cleared": float(per_hr_cleared.mean()),
            "no_doubters": int((per_hr_cleared == matrix.cleared.shape[1]).sum()),
            "cheap_hrs": int((per_hr_cleared <= 5).sum()),
        }
    ```
